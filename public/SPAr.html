Component Method Navigation

Define navigation methods in your components and call them via HTML.patch.

Component Method Navigation Examples

-- Component with navigation methods
local UserListComponent = FunctionalComponent:extends()
UserListComponent:setReactiveView()

-- Define component methods
UserListComponent.methods.navigateToUserProfile = function(self, ws_id, args)
    local userId = args.userId
    local parent = self:getParentComponent()
    
    if parent and parent.methods.navigateTo then
        parent.methods.navigateTo(parent, ws_id, {
            routeName = "userProfile",
            params = { userId = userId }
        })
    end
end

UserListComponent.methods.showCreateUserModal = function(self, ws_id, args)
    local parent = self:getParentComponent()
    
    if parent and parent.methods.showModal then
        parent.methods.showModal(parent, ws_id, {
            modalType = "userForm",
            props = { mode = "create" }
        })
    end
end

function UserListComponent:renderLayout(state, props, children, H)
    local users = state.users or {}
    local parentKey = self:getParentKey()
    local componentKey = self:getComponentKey()
    
    return H.e("div", { class = "p-6" }, {
        H.e("div", { class = "flex justify-between items-center mb-6" }, {
            H.e("h1", { class = "text-2xl font-bold" }, "Users"),
            H.e("button", {
                class = "px-4 py-2 bg-blue-500 text-white rounded",
                onClick = H.patch(parentKey, componentKey, "showCreateUserModal", {})
            }, "Add User")
        }),
        
        H.e("div", { class = "space-y-4" }, 
            self:map(users, function(user, index)
                return H.e("div", { 
                    class = "p-4 border rounded-lg bg-white shadow-sm",
                    key = user.id
                }, {
                    H.e("h3", { class = "font-semibold" }, user.name),
                    H.e("p", { class = "text-gray-600" }, user.email),
                    H.e("button", {
                        class = "mt-2 px-3 py-1 bg-blue-100 text-blue-700 rounded text-sm",
                        onClick = H.patch(parentKey, componentKey, "navigateToUserProfile", {
                            userId = user.id
                        })
                    }, "View Profile")
                })
            end)
        )
    })
end

-- Advanced component with multiple navigation methods
local ProductCatalogComponent = FunctionalComponent:extends()
ProductCatalogComponent:setReactiveView()

-- Navigation methods
ProductCatalogComponent.methods.navigateToProduct = function(self, ws_id, args)
    local parent = self:getParentComponent()
    parent.methods.navigateTo(parent, ws_id, {
        routeName = "productDetail",

            categoryId = args.categoryId,
            productId = args.productId
        },
        config = {
            animation = "slide",
            preserveState = true
        }
    })
end

ProductCatalogComponent.methods.navigateToCategory = function(self, ws_id, args)
    local parent = self:getParentComponent()
    parent.methods.navigateTo(parent, ws_id, {
        routeName = "category",
        params = { categoryId = args.categoryId }
    })
end

ProductCatalogComponent.methods.navigateBack = function(self, ws_id, args)
    local parent = self:getParentComponent()
    parent.methods.navigateBack(parent, ws_id, {})
end

function ProductCatalogComponent:renderLayout(state, props, children, H)
    local products = state.products or {}
    local categories = state.categories or {}
    local parentKey = self:getParentKey()
    local componentKey = self:getComponentKey()
    
    return H.e("div", { class = "p-6" }, {
        -- Breadcrumb navigation
        H.e("nav", { class = "mb-6 flex space-x-2 text-sm" }, {
            H.e("button", {
                class = "text-blue-500 hover:underline",
                onClick = H.patch(parentKey, componentKey, "navigateBack", {})
            }, "← Back"),
            H.e("span", {}, " / "),
            H.e("span", { class = "text-gray-600" }, "Catalog")
        }),
        
        -- Category filters
        H.e("div", { class = "mb-6 flex flex-wrap gap-2" },
            self:map(categories, function(category)
                return H.e("button", {
                    class = "px-3 py-1 bg-gray-200 rounded-full text-sm",
                    onClick = H.patch(parentKey, componentKey, "navigateToCategory", {
                        categoryId = category.id
                    })
                }, category.name)
            end)
        ),
        
        -- Product grid
        H.e("div", { class = "grid grid-cols-1 md:grid-cols-3 gap-6" },
            self:map(products, function(product)
                return H.e("div", {
                    class = "border rounded-lg overflow-hidden cursor-pointer hover:shadow-lg transition-shadow",
                    onClick = H.patch(parentKey, componentKey, "navigateToProduct", {
                        categoryId = product.categoryId,
                        productId = product.id
                    })
                }, {
                    H.e("img", {
                        src = product.image,
                        class = "w-full h-48 object-cover"
                    }),
                    H.e("div", { class = "p-4" }, {
                        H.e("h3", { class = "font-semibold text-lg" }, product.name),
                        H.e("p", { class = "text-gray-600 mt-2" }, product.description),
                        H.e("div", { class = "mt-3 flex justify-between items-center" }, {
                            H.e("span", { class = "text-green-600 font-bold" }, 
                                "$" .. product.price),
                            H.e("button", {
                                class = "px-3 py-1 bg-blue-500 text-white rounded text-sm",
                                onClick = H.patch(parentKey, componentKey, "quickAddToCart", {
                                    productId = product.id
                                })
                            }, "Add to Cart")
                        })
                    })
                })
            end)
        )
    })
end

Modal Management with Component Methods

Manage modals through component methods using HTML.patch for event handling.

Modal Component with Method Handlers

-- Modal component using component methods
local ConfirmModalComponent = FunctionalComponent:extends()
ConfirmModalComponent:setReactiveView()

-- Modal methods
ConfirmModalComponent.methods.handleConfirm = function(self, ws_id, args)
    local onConfirm = self.props.onConfirm
    local modalId = self.props.modalId
    
    if type(onConfirm) == "function" then
        onConfirm(args)
    end
    
    -- Close modal
    local parent = self:getParentComponent()
    if parent and parent.methods.hideModal then
        parent.methods.hideModal(parent, ws_id, { modalId = modalId })
    end
end

ConfirmModalComponent.methods.handleCancel = function(self, ws_id, args)
    local onCancel = self.props.onCancel
    local modalId = self.props.modalId
    
    if type(onCancel) == "function" then
        onCancel(args)
    end
    
    -- Close modal
    local parent = self:getParentComponent()
    if parent and parent.methods.hideModal then
        parent.methods.hideModal(parent, ws_id, { modalId = modalId })
    end
end

function ConfirmModalComponent:renderLayout(state, props, children, H)
    local message = props.message or "Are you sure?"
    local confirmText = props.confirmText or "Confirm"
    local cancelText = props.cancelText or "Cancel"
    local parentKey = self:getParentKey()
    local componentKey = self:getComponentKey()
    
    return H.e("div", { 
        class = "modal-content bg-white p-6 rounded-lg shadow-xl max-w-md mx-auto"
    }, {
        H.e("h2", { class = "text-xl font-bold mb-4" }, message),
        H.e("div", { class = "flex justify-end space-x-4" }, {
            H.e("button", {
                class = "px-4 py-2 bg-gray-300 rounded hover:bg-gray-400",
                onClick = H.patch(parentKey, componentKey, "handleCancel", {})
            }, cancelText),
            H.e("button", {
                class = "px-4 py-2 bg-red-500 text-white rounded hover:bg-red-600",
                onClick = H.patch(parentKey, componentKey, "handleConfirm", {})
            }, confirmText)
        })
    })
end

Complex Modal Workflow with Methods

-- Multi-step form modal with component methods
local UserFormModalComponent = FunctionalComponent:extends()
UserFormModalComponent:setReactiveView()

-- Form state management
UserFormModalComponent.methods.updateFormField = function(self, ws_id, args)
    local field = args.field
    local value = args.value
    
    self:setState(function(state)
        local formData = state.formData or {}
        formData[field] = value
        return { formData = formData }
    end)
end

UserFormModalComponent.methods.nextStep = function(self, ws_id, args)
    self:setState(function(state)
        local currentStep = state.currentStep or 1
        return { currentStep = currentStep + 1 }
    end)
end

UserFormModalComponent.methods.prevStep = function(self, ws_id, args)
    self:setState(function(state)
        local currentStep = state.currentStep or 1
        return { currentStep = math.max(1, currentStep - 1) }
    end)
end

UserFormModalComponent.methods.submitForm = function(self, ws_id, args)
    local formData = self.state.formData or {}
    local onSuccess = self.props.onSuccess
    local modalId = self.props.modalId
    
    -- Validate form data
    if not formData.name or formData.name:trim() == "" then
        self:setState({ error = "Name is required" })
        return
    end
    
    -- Submit data
    if type(onSuccess) == "function" then
        onSuccess(formData)
    end
    
    -- Close modal
    local parent = self:getParentComponent()
    if parent and parent.methods.hideModal then
        parent.methods.hideModal(parent, ws_id, { modalId = modalId })
    end
end

function UserFormModalComponent:renderLayout(state, props, children, H)
    local currentStep = state.currentStep or 1
    local formData = state.formData or {}
    local error = state.error
    local parentKey = self:getParentKey()
    local componentKey = self:getComponentKey()
    
    return H.e("div", { 
        class = "modal-content bg-white p-6 rounded-lg shadow-xl max-w-2xl mx-auto"
    }, {
        -- Progress steps
        H.e("div", { class = "flex justify-between mb-6" }, {
            H.e("div", { 
                class = "flex-1 text-center " .. (currentStep >= 1 and "text-blue-600 font-bold" or "text-gray-400")
            }, "1. Basic Info"),
            H.e("div", { 
                class = "flex-1 text-center " .. (currentStep >= 2 and "text-blue-600 font-bold" or "text-gray-400")
            }, "2. Details"),
            H.e("div", { 
                class = "flex-1 text-center " .. (currentStep >= 3 and "text-blue-600 font-bold" or "text-gray-400")
            }, "3. Review")
        }),
        
        -- Error message
        error and H.e("div", { class = "mb-4 p-3 bg-red-100 text-red-700 rounded" }, error),
        
        -- Step 1: Basic Information
        currentStep == 1 and H.e("div", { class = "space-y-4" }, {
            H.e("h3", { class = "text-lg font-semibold" }, "Basic Information"),
            H.e("input", {
                type = "text",
                placeholder = "Full Name",
                value = formData.name or "",
                class = "w-full p-2 border rounded",
                onInput = H.patch(parentKey, componentKey, "updateFormField", {
                    field = "name",
                    value = "__VALUE__"
                })
            }),
            H.e("input", {
                type = "email",
                placeholder = "Email Address",
                value = formData.email or "",
                class = "w-full p-2 border rounded",
                onInput = H.patch(parentKey, componentKey, "updateFormField", {
                    field = "email",
                    value = "__VALUE__"
                })
            }),
            H.e("button", {
                class = "w-full bg-blue-500 text-white py-2 rounded",
                onClick = H.patch(parentKey, componentKey, "nextStep", {})
            }, "Next")
        }),
        
        -- Step 2: Additional Details
        currentStep == 2 and H.e("div", { class = "space-y-4" }, {
            H.e("h3", { class = "text-lg font-semibold" }, "Additional Details"),
            H.e("textarea", {
                placeholder = "Bio",
                value = formData.bio or "",
                class = "w-full p-2 border rounded h-24",
                onInput = H.patch(parentKey, componentKey, "updateFormField", {
                    field = "bio",
                    value = "__VALUE__"
                })
            }),
            H.e("div", { class = "flex space-x-4" }, {
                H.e("button", {
                    class = "flex-1 bg-gray-300 py-2 rounded",
                    onClick = H.patch(parentKey, componentKey, "prevStep", {})
                }, "Back"),
                H.e("button", {
                    class = "flex-1 bg-blue-500 text-white py-2 rounded",
                    onClick = H.patch(parentKey, componentKey, "nextStep", {})
                }, "Next")
            })
        }),
        
        -- Step 3: Review and Submit
        currentStep == 3 and H.e("div", { class = "space-y-4" }, {
            H.e("h3", { class = "text-lg font-semibold" }, "Review Information"),
            H.e("div", { class = "space-y-2" }, {
                H.e("p", {}, "Name: " .. (formData.name or "Not provided")),
                H.e("p", {}, "Email: " .. (formData.email or "Not provided")),
                H.e("p", {}, "Bio: " .. (formData.bio or "Not provided"))
            }),
            H.e("div", { class = "flex space-x-4" }, {
                H.e("button", {
                    class = "flex-1 bg-gray-300 py-2 rounded",
                    onClick = H.patch(parentKey, componentKey, "prevStep", {})
                }, "Back"),
                H.e("button", {
                    class = "flex-1 bg-green-500 text-white py-2 rounded",
                    onClick = H.patch(parentKey, componentKey, "submitForm", {})
                }, "Submit")
            })
        })
    })
end

Complete integration example using component methods for all interactions.

Main Application with Method Architecture

-- Main application component with method-based architecture
local MainApp = FunctionalComponent:extends()
MainApp:setReactiveView()

-- Application-level methods
MainApp.methods.handleUserAction = function(self, ws_id, args)
    local action = args.action
    local data = args.data
    
    if action == "userCreated" then
        -- Refresh user list or show success message
        self:setState({ 
            lastAction = "userCreated",
            lastActionData = data
        })
        
        -- Show success modal
        self.methods.showModal(self, ws_id, {
            modalType = "alert",
            props = {
                message = "User created successfully!",
                type = "success"
            }
        })
    end
end

function MainApp:renderLayout(state, props, children, H, client_state)
    local parentKey = self:getParentKey()
    local componentKey = self:getComponentKey()
    
    return H.e("div", { class = "app-container" }, {
        -- Header with navigation
        H.e("header", { class = "bg-white shadow-sm p-4" }, {
            H.e("div", { class = "flex justify-between items-center" }, {
                H.e("h1", { 
                    class = "text-xl font-bold cursor-pointer",
                    onClick = H.patch(parentKey, componentKey, "navigateToHome", {})
                }, "My App"),
                H.e("nav", { class = "flex space-x-4" }, {
                    H.e("button", {
                        class = "text-blue-500 hover:underline",
                        onClick = H.patch(parentKey, componentKey, "navigateToUsers", {})
                    }, "Users"),
                    H.e("button", {
                        class = "text-blue-500 hover:underline",
                        onClick = H.patch(parentKey, componentKey, "navigateToSettings", {})
                    }, "Settings")
                })
            })
        }),
        
        -- Main content area
        H.e("main", { class = "flex-1" }, {
            HTML.Component("routeManager", {}),
            HTML.Component("modalManager", {})
        }),
        
        -- Footer
        H.e("footer", { class = "bg-gray-100 p-4 text-center" }, {
            H.e("p", { class = "text-gray-600" }, "© 2023 My App. All rights reserved.")
        })
    })
end

-- Navigation methods
MainApp.methods.navigateToHome = function(self, ws_id, args)
    self:callChildMethod("routeManager", "navigateTo", ws_id, {
        routeName = "home"
    })
end

MainApp.methods.navigateToUsers = function(self, ws_id, args)
    self:callChildMethod("routeManager", "navigateTo", ws_id, {
        routeName = "users"
    })
end

MainApp.methods.navigateToSettings = function(self, ws_id, args)
    self:callChildMethod("routeManager", "navigateTo", ws_id, {
        routeName = "settings"
    })
end

-- Modal methods
MainApp.methods.showModal = function(self, ws_id, args)
    self:callChildMethod("modalManager", "showModal", ws_id, args)
end

MainApp.methods.hideModal = function(self, ws_id, args)
    self:callChildMethod("modalManager", "hideModal", ws_id, args)
end

return MainApp
